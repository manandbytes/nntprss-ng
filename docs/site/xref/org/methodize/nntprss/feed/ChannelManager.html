96      public void configure(Document config) {
97  
98          channelDAO.loadConfiguration(this);
99  
100         updateProxyConfig();
101 
102 		// Get poller Configuration
103 		Element rootElm = config.getDocumentElement();
104 		NodeList cfgElms = rootElm.getElementsByTagName("poller");
105 		if(cfgElms != null && cfgElms.getLength() > 0)
106 		{
107 			Element pollerElm = (Element)cfgElms.item(0);
108 			String threadsStr = pollerElm.getAttribute("threads");
109 			if(threadsStr != null) {
110 				pollerThreads = Integer.parseInt(threadsStr);
111 			}
112 		}
113 
114 		NodeList processorsElms = rootElm.getElementsByTagName("itemProcessors");
115 		if(processorsElms != null && processorsElms.getLength() > 0) {
116 			NodeList processorElms = ((Element)processorsElms.item(0)).getElementsByTagName("processor");
117 			if(processorElms != null && processorElms.getLength() > 0) {
118 				List processors = new ArrayList();
119 				for(int i = 0; i < processorElms.getLength(); i++) {
120 					Element processorElm = (Element)processorElms.item(i);
121 
122 					String itemProcessorClassName = processorElm.getAttribute("class");
123 					if(itemProcessorClassName != null){
124 						try {
125 							Object itemProcessorObject = Class.forName(itemProcessorClassName).newInstance();
126 							if(!(itemProcessorObject instanceof ItemProcessor))
127 							{
128 								log.warn(itemProcessorClassName + " not instance of org.methodize.nntprss.plugin.ItemProcessor, skipping");
129 							}
130 							else
131 							{
132 								try {
133 									ItemProcessor itemProcessor = (ItemProcessor)itemProcessorObject;
134 									itemProcessor.initialize(processorElm);
135 									processors.add(itemProcessor);
136 								} catch(PluginException pe) {
137 									log.warn("Error initializing ItemProcessor plug-in: " + pe.getMessage() + ", skipping.", pe);
138 								}
139 							}
140 						} catch(ClassNotFoundException cnfe) {
141 							log.warn("Cannot find ItemProcessor class " + itemProcessorClassName, cnfe);
142 						} catch(IllegalAccessException iae) {
143 							log.warn("Error instantiating ItemProcessor class: " + iae.getMessage(), iae);
144 						} catch(InstantiationException ie) {
145 							log.warn("Error instantiating ItemProcessor class: " + ie.getMessage(), ie);
146 						}
147 					}
148 					
149 				}
150 				if(processors.size() > 0)
151 				{
152 					itemProcessors = (ItemProcessor[])processors.toArray(new ItemProcessor[0]);
153 				}
154 			}
155 		}
156 
157 		// Load feeds...
158 		categories = channelDAO.loadCategories();
159 		channels = channelDAO.loadChannels(this);
160     }

228     public Category categoryById(int id) {
229         Category category = null;
230         if(categories != null) 
231         {
232 	        Iterator categoryIter = categories.values().iterator();
233 	        while (categoryIter.hasNext()) {
234 	            Category nextCategory = (Category) categoryIter.next();
235 	            if (nextCategory.getId() == id) {
236 	                category = nextCategory;
237 	                break;
238 	            }
239 	        }
240 		}
241         return category;
242     }

255     public synchronized void repollAllChannels() {
256         try {
257             Iterator channelIter = channels.values().iterator();
258 
259             while (channelIter.hasNext()) {
260                 Channel channel = (Channel) channelIter.next();
261                 channel.setLastPolled(null);
262                 channel.setStatus(Channel.STATUS_OK);
263             }
264 
265         } catch (ConcurrentModificationException cme) {
266             // Just in case something else is modifying the channel structure...
267         }
268     }

270     public void configureHttpClient(HttpClient client) {
271         client.setHostConfiguration(hostConfig);
272 
273         if ((proxyUserID != null && proxyUserID.length() > 0)
274             || (proxyPassword != null && proxyPassword.length() > 0)) {
275             client.getState().setProxyCredentials(
276                 null,
277                 null,
278                 new UsernamePasswordCredentials(
279                     proxyUserID,
280                     (proxyPassword == null) ? "" : proxyPassword));
281         } else {
282             client.getState().setProxyCredentials(null, null, null);
283         }
284     }

286     private void updateProxyConfig() {
287         // Set proxy configuration, if necessary.
288         if (useProxy && (proxyServer != null) && (proxyServer.length() > 0)) {
289             // Set HttpClient proxy configuration
290             hostConfig.setProxy(proxyServer, proxyPort);
291         } else {
292             hostConfig = new HostConfiguration();
293         }
294     }

296     public void saveConfiguration() {
297         channelDAO.saveConfiguration(this);

300         updateProxyConfig();
