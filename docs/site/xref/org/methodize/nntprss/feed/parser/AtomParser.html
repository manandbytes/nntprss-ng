115     @Override
116     public void extractFeedInfo(Element docRootElement, Channel channel) {
117         // Read header...
118         channel.setTitle(
119             XMLHelper.getChildElementValue(docRootElement, "title"));
120         // XXX Currently assign channelTitle to author
121         channel.setAuthor(channel.getTitle());
122 
123     	URI xmlBase = extractXmlBase(docRootElement, channel);
124         channel.setLink(extractLink(channel, docRootElement, xmlBase));
125 
126         // @TODO: Summary or Subtitle?
127         String description;
128         description = XMLHelper.getChildElementValue(docRootElement, "summary");
129         if (description == null || description.length() == 0) {
130             description =
131                 XMLHelper.getChildElementValue(docRootElement, "subtitle");
132         }
133         channel.setDescription(description);
134 
135         // Build author...
136         NodeList authorList = docRootElement.getElementsByTagName("author");
137         if (authorList.getLength() > 0) {
138             Element authorElm = (Element) authorList.item(0);
139             channel.setManagingEditor(extractAuthor(authorElm));
140         } else {
141             channel.setManagingEditor(null);
142         }
143     }

145     private String extractAuthor(Element authorElm) {
146         StringBuffer author = new StringBuffer();
147         String name = XMLHelper.getChildElementValue(authorElm, "name");
148         String email = XMLHelper.getChildElementValue(authorElm, "email");
149         if (name != null && name.length() > 0) {
150             author.append(name);
151         }
152 
153         if (email != null && email.length() > 0) {
154             if (author.length() > 0) {
155                 author.append(" (").append(email).append(")");
156             } else {
157                 author.append(email);
158             }
159         }
160         return author.toString();
161     }

163     @Override
164     public void processFeedItems(
165         Element rootElm,
166         Channel channel,
167         ChannelDAO channelDAO,
168         boolean keepHistory)
169         throws NoSuchAlgorithmException, IOException {
170 
171     	URI xmlBase = extractXmlBase(rootElm, channel);
172     	
173         NodeList entryList = rootElm.getElementsByTagName("entry");
174 
175         Calendar retrievalDate = Calendar.getInstance();
176         retrievalDate.add(Calendar.SECOND, -entryList.getLength());
177 
178         Set currentSignatures = null;
179         if (!keepHistory) {
180             currentSignatures = new HashSet();
181         }
182 
183         Map newItems = new HashMap();
184         Map newItemKeys = new HashMap();
185         // orderedItems maintains original document first-to-last order
186         // Assumption: Items in the Atom document go from most recent
187         // to earliest.  This is used to assign date/times in a reasonably
188         // sensible order to those feeds that do not provide a date
189 
190         List orderedItems = new ArrayList();
191 
192         // Calculate signature
193         MessageDigest md = MessageDigest.getInstance("MD5");
194 
195         for (int itemCount = entryList.getLength() - 1;
196             itemCount >= 0;
197             itemCount--) {
198             Element itemElm = (Element) entryList.item(itemCount);
199             String signatureStr = generateItemSignature(md, itemElm);
200 
201             if (!keepHistory) {
202                 currentSignatures.add(signatureStr);
203             }
204             newItems.put(signatureStr, itemElm);
205             newItemKeys.put(itemElm, signatureStr);
206             orderedItems.add(itemElm);
207         }
208 
209         if (newItems.size() > 0) {
210             // Discover new items...
211             Set newItemSignatures =
212                 channelDAO.findNewItemSignatures(channel, newItems.keySet());
213 
214             if (newItemSignatures.size() > 0) {
215                 for (int i = 0; i < orderedItems.size(); i++) {
216                     Element entryElm = (Element) orderedItems.get(i);
217                     String signatureStr = (String) newItemKeys.get(entryElm);
218 
219                     // If signature is not in new items set, skip...
220                     if (!newItemSignatures.contains(signatureStr))
221                         continue;
222 
223                     String title =
224                         XMLHelper.getChildElementValue(entryElm, "title", "");
225                     //					String link =
226                     //						XMLHelper.getChildElementValue(entryElm, "link", "");
227                     String link = extractLink(channel, entryElm, xmlBase);
228 
229                     String guid =
230                         XMLHelper.getChildElementValue(entryElm, "id");
231                     boolean guidIsPermaLink = false;
232                     //					if (guid != null) {
233                     //						String guidIsPermaLinkStr =
234                     //							XMLHelper.getChildElementAttributeValue(
235                     //								itemElm,
236                     //								"guid",
237                     //								"guidIsPermaLink");
238                     //						if (guidIsPermaLinkStr != null) {
239                     //							guidIsPermaLink =
240                     //								guidIsPermaLinkStr.equalsIgnoreCase("true");
241                     //						}
242                     //					}
243 
244                     // Handle xhtml:body / content:encoded / description
245                     String description = processContent(entryElm);
246 
247                     // @TODO: Comments in Atom?
248                     String comments = null;
249 
250                     // Date - use modified, if not present use created
251                     String pubDateStr =
252                         XMLHelper.getChildElementValue(entryElm, "modified");
253                     if (pubDateStr == null || pubDateStr.length() == 0) {
254                         pubDateStr =
255                             XMLHelper.getChildElementValue(entryElm, "created");
256                     }
257 
258                     Date pubDate = null;
259 
260                     if (pubDateStr != null && pubDateStr.length() > 0) {
261                         log.debug("create/modified == " + pubDateStr);
262                         SimpleDateFormat[] dateParserArray =
263                             (SimpleDateFormat[]) dateParsers.get();
264                         for (int parseCount = 0;
265                             parseCount < dateParserArray.length;
266                             parseCount++) {
267                             try {
268                                 pubDate =
269                                     dateParserArray[parseCount].parse(
270                                         pubDateStr);
271                             } catch (ParseException pe) {
272                             }
273                             if (pubDate != null)
274                                 break;
275                         }
276                         if (pubDate != null) {
277                             log.debug("processed Atom feed date == " + pubDate);
278                         } else {
279                             log.debug(
280                                 "Invalid Atom feed date format - "
281                                     + pubDateStr);
282                         }
283                     }
284 
285                     String creator = null;
286                     NodeList authorList =
287                         entryElm.getElementsByTagName("author");
288                     if (authorList.getLength() > 0) {
289                         Element authorElm = (Element) authorList.item(0);
290                         creator = extractAuthor(authorElm);
291                     }
292 
293                     int lastArticleNumber = channel.getLastArticleNumber();
294                     Item item = new Item(++lastArticleNumber, signatureStr);
295                     channel.setLastArticleNumber(lastArticleNumber);
296 
297                     item.setChannel(channel);
298 
299                     if (title.length() > 0) {
300                         item.setTitle(title);
301                     } else {
302                         // We need to create a initial title from the description, because
303                         // we do have a description, don't we???
304                         String strippedDesc =
305                             stripControlChars(XMLHelper.stripTags(description));
306                         int length =
307                             strippedDesc.length() > 64
308                                 ? 64
309                                 : strippedDesc.length();
310                         item.setTitle(strippedDesc.substring(0, length));
311                     }
312                     item.setDescription(description);
313                     item.setLink(link);
314                     item.setGuid(guid);
315                     item.setGuidIsPermaLink(guidIsPermaLink);
316                     item.setComments(comments);
317                     item.setCreator(creator);
318 
319                     if (pubDate == null) {
320                         item.setDate(retrievalDate.getTime());
321                         // Add 1 second - to introduce some distinction date-wise
322                         // between items
323                         retrievalDate.add(Calendar.SECOND, 1);
324                     } else {
325                         item.setDate(pubDate);
326                     }
327 
328 					// Invoke ItemProcessor
329 					invokeProcessors(item);
330 
331                     // persist to database...
332                     channelDAO.saveItem(item);
333                     channel.setTotalArticles(channel.getTotalArticles() + 1);
334                 }
335             }
336 
337         }
338 
339         if (!keepHistory) {
340             if (currentSignatures.size() > 0) {
341                 if (channel.getExpiration() == 0) {
342                     channelDAO.deleteItemsNotInSet(channel, currentSignatures);
343 					channel.setTotalArticles(currentSignatures.size());
344                 } else if (
345                     channel.getExpiration() > 0
346 						&& (channel.getLastCleaned() == null || channel.getLastCleaned().before(
347 					new Date(
348 						System.currentTimeMillis()
349 							- Channel.CLEANING_INTERVAL)))) {
350                     channelDAO.deleteExpiredItems(channel, currentSignatures);
351                     channel.setLastCleaned(new Date());
352                 }
353             }
354         }
355     }

361 	private URI extractXmlBase(Element rootElm, Channel channel, URI existingBase)
362 			 {
363 		URI xmlBase;
364 		// Handle xml:base
365     	Attr xmlBaseAttr = rootElm.getAttributeNodeNS(XMLHelper.XML_NS_URI, "base");
366 
367     	// XXX Malformed URLs?
368     	try {
369 	    	if(xmlBaseAttr != null) {
370 	    		if(existingBase == null) {
371 	    			xmlBase = new URI(xmlBaseAttr.getValue());
372 	    		} else {
373 	    			xmlBase = existingBase.resolve(xmlBaseAttr.getValue());
374 	    		}
375 	    	} else {
376 	    		if(existingBase != null) {
377 	    			xmlBase = existingBase;
378 	    		} else {
379 	    			xmlBase = new URI(channel.getUrl());
380 	    		}
381 	    	}
382     	}
383     	catch(URISyntaxException use) {
384     		// XXX Use channel URL - this should be ok, otherwise
385     		// we would have not got here in the first place
386     		try {
387     			if(existingBase != null) {
388     				xmlBase = existingBase;
389     			} else {
390     				xmlBase = new URI(channel.getUrl());
391     			}
392     		}
393     		catch(URISyntaxException use2) { 
394     			xmlBase = null;
395     		}
396     	}
397 		return xmlBase;
398 	}

400 	@Override
401     public String processContent(Element itemElm) {
402         // Check for xhtml:body
403         String description = null;
404 
405         NodeList contentList = itemElm.getElementsByTagName("content");
406 
407         if (contentList.getLength() > 0) {
408             Element contentElm = (Element) contentList.item(0);
409 
410             String type = contentElm.getAttribute("type");
411             String mode = contentElm.getAttribute("mode");
412             if ((type.startsWith("application/xhtml")
413                 && !mode.equalsIgnoreCase("escaped")) || 
414                 type.equals("xhtml")) {
415                 // xhtml body 				
416                 NodeList children = contentElm.getChildNodes();
417                 StringBuffer content = new StringBuffer();
418                 for (int childCount = 0;
419                     childCount < children.getLength();
420                     childCount++) {
421                     content.append(children.item(childCount).toString());
422                 }
423                 description = content.toString();
424             } else {
425                 // escaped text...				
426                 description = XMLHelper.getElementValue(contentElm);
427             }
428         }
429 
430         // If all else fails, drop back and use the contents of the summary...
431         if ((description == null) || (description.length() == 0)) {
432             description =
433                 XMLHelper.getChildElementValue(itemElm, "summary", "");
434         }
435         return description;
436     }

438     private String extractLink(Channel channel, Element itemElm, URI xmlBase) {
439         String link = null;
440 
441         // Get entry's XML base
442         URI entryXmlBase = extractXmlBase(itemElm, channel, xmlBase);
443         NodeList elemList = itemElm.getChildNodes();
444         for (int i = 0; i < elemList.getLength(); i++) {
445             if (elemList.item(i).getNodeName().equals("link")
446                 &&
447                 ( elemList.item(i).getNamespaceURI().equals(XMLNS_ATOM) ||
448                 elemList.item(i).getNamespaceURI().equals(XMLNS_OLD_ATOM)) ) {
449                 Element linkElm = (Element) elemList.item(i);
450                 Attr linkRelNode = linkElm.getAttributeNode("rel");
451                 if(linkRelNode == null || linkRelNode.getValue().equals("alternate")) {
452                     String entryLink = linkElm.getAttribute("href");
453                     link = entryXmlBase.resolve(entryLink).toString();
454                     break;
455                 }
456             }
457         }
458         return link;
