106     public boolean isParsable(Element docRootElement) {
107         if (docRootElement.getNodeName().equals("rss")
108             || docRootElement.getNodeName().equals("rdf:RDF")) {
109             return true;
110         } else {
111             return false;
112         }
113     }
114 
115     public String getFormatVersion(Element docRootElement) {
116         String rssVersion;
117 
118         if (docRootElement.getNodeName().equals("rss")) {
119             rssVersion = "RSS " + docRootElement.getAttribute("version");
120         } else if (docRootElement.getNodeName().equals("rdf:RDF")) {
121             rssVersion = "RDF";
122         } else {
123             rssVersion = "Unknown RSS";
124         }
125         return rssVersion;
126     }

129         Element channelElm =

143     public void processFeedItems(
144         Element rootElm,
145         Channel channel,
146         ChannelDAO channelDAO,
147         boolean keepHistory)
148         throws NoSuchAlgorithmException, IOException {
149         Element rssDocElm =
150             (Element) rootElm.getElementsByTagName("channel").item(0);
151         // Check for items within channel element and outside 
152         // channel element
153         NodeList itemList = rssDocElm.getElementsByTagName("item");
154 
155         if (itemList.getLength() == 0) {
156             itemList = rootElm.getElementsByTagName("item");
157         }
158 
159         Calendar retrievalDate = Calendar.getInstance();
160         retrievalDate.add(Calendar.SECOND, -itemList.getLength());
161 
162         Set currentSignatures = null;
163         if (!keepHistory) {
164             currentSignatures = new HashSet();
165         }
166 
167         Map newItems = new HashMap();
168         Map newItemKeys = new HashMap();
169         // orderedItems maintains original document first-to-last order
170         // Assumption: Items in the RSS document go from most recent
171         // to earliest.  This is used to assign date/times in a reasonably
172         // sensible order to those feeds that do not provide either pubDate
173         // or dc:date
174 
175         List orderedItems = new ArrayList();
176 
177         // Calculate signature
178         MessageDigest md = MessageDigest.getInstance("MD5");
179 
180         for (int itemCount = itemList.getLength() - 1;
181             itemCount >= 0;
182             itemCount--) {
183             Element itemElm = (Element) itemList.item(itemCount);
184             String signatureStr = generateItemSignature(md, itemElm);
185 
186             if (!keepHistory) {
187                 currentSignatures.add(signatureStr);
188             }
189             newItems.put(signatureStr, itemElm);
190             newItemKeys.put(itemElm, signatureStr);
191             orderedItems.add(itemElm);
192         }
193 
194         if (newItems.size() > 0) {
195             // Discover new items...
196             Set newItemSignatures =
197                 channelDAO.findNewItemSignatures(channel, newItems.keySet());
198 
199             if (newItemSignatures.size() > 0) {
200                 for (int i = 0; i < orderedItems.size(); i++) {
201                     Element itemElm = (Element) orderedItems.get(i);
202                     String signatureStr = (String) newItemKeys.get(itemElm);
203 
204                     // If signature is not in new items set, skip...
205                     if (!newItemSignatures.contains(signatureStr))
206                         continue;
207 
208                     String title =
209                         XMLHelper.getChildElementValue(itemElm, "title", "");
210                     String link =
211                         XMLHelper.getChildElementValue(itemElm, "link", "");
212 
213                     String guid =
214                         XMLHelper.getChildElementValue(itemElm, "guid");
215                     boolean guidIsPermaLink = true;
216                     if (guid != null) {
217                         String guidIsPermaLinkStr =
218                             XMLHelper.getChildElementAttributeValue(
219                                 itemElm,
220                                 "guid",
221                                 "isPermaLink");
222                         if (guidIsPermaLinkStr != null) {
223                             guidIsPermaLink =
224                                 guidIsPermaLinkStr.equalsIgnoreCase("true");
225                         }
226                     }
227 
228                     // Handle xhtml:body / content:encoded / description
229                     String description = processContent(itemElm);
230 
231                     String comments =
232                         XMLHelper.getChildElementValue(itemElm, "comments", "");
233 
234                     Date pubDate = null;
235                     String pubDateStr =
236                         XMLHelper.getChildElementValue(itemElm, "pubDate");
237                     if (pubDateStr != null && pubDateStr.length() > 0) {
238                         // Parse Date...
239                         log.debug("pubDate == " + pubDateStr);
240                         try {
241                             pubDate =
242                                 ((MailDateFormat) date822Parser.get()).parse(
243                                     pubDateStr);
244                             log.debug("processed pubDate == " + pubDate);
245                         } catch (ParseException pe) {
246                             log.debug("Invalid pubDate format - " + pubDateStr);
247                         }
248                     }
249 
250                     if (pubDate == null) {
251                         // Try for Dublin Core Date
252                         String dcDateStr =
253                             XMLHelper.getChildElementValueNS(
254                                 itemElm,
255                                 RSSHelper.XMLNS_DC,
256                                 "date");
257                         if (dcDateStr != null && dcDateStr.length() > 0) {
258                             log.debug("dc:date == " + dcDateStr);
259 
260                             if (dcDateStr.indexOf("GMT") == -1) {
261                                 // Check for : in RFC822 time zone...
262                                 int hourColon = dcDateStr.indexOf(":");
263                                 if (hourColon != -1) {
264                                     int minuteColon =
265                                         dcDateStr.indexOf(":", hourColon + 1);
266                                     if (minuteColon != -1) {
267                                         int timeZoneColon =
268                                             dcDateStr.indexOf(
269                                                 ":",
270                                                 minuteColon + 1);
271                                         if (timeZoneColon != -1) {
272                                             if (dcDateStr.length()
273                                                 > timeZoneColon) {
274                                                 dcDateStr =
275                                                     dcDateStr.substring(
276                                                         0,
277                                                         timeZoneColon)
278                                                         + dcDateStr.substring(
279                                                             timeZoneColon + 1);
280                                             }
281                                         }
282                                     }
283                                 }
284 
285                             }
286 
287                             SimpleDateFormat[] dcDateArray =
288                                 (SimpleDateFormat[]) dcDates.get();
289                             for (int parseCount = 0;
290                                 parseCount < dcDateArray.length;
291                                 parseCount++) {
292                                 try {
293                                     pubDate =
294                                         dcDateArray[parseCount].parse(
295                                             dcDateStr);
296                                 } catch (ParseException pe) {
297                                 }
298                                 if (pubDate != null)
299                                     break;
300                             }
301                             if (pubDate != null) {
302                                 log.debug("processed dc:date == " + pubDate);
303                             } else {
304                                 log.debug(
305                                     "Invalid dc:date format - " + dcDateStr);
306                             }
307                         }
308                     }
309 
310                     String dcCreator =
311                         XMLHelper.getChildElementValueNS(
312                             itemElm,
313                             RSSHelper.XMLNS_DC,
314                             "creator");
315 
316                     int lastArticleNumber = channel.getLastArticleNumber();
317                     Item item = new Item(++lastArticleNumber, signatureStr);
318                     channel.setLastArticleNumber(lastArticleNumber);
319 
320                     item.setChannel(channel);
321 
322                     if (title.length() > 0) {
323                         item.setTitle(title);
324                     } else {
325                         // We need to create a initial title from the description, because
326                         // we do have a description, don't we???
327                         String strippedDesc =
328                             stripControlChars(XMLHelper.stripTags(description));
329                         int length =
330                             strippedDesc.length() > 64
331                                 ? 64
332                                 : strippedDesc.length();
333                         item.setTitle(strippedDesc.substring(0, length));
334                     }
335                     
336                     // If empty description, set the description to the title.
337                     // This will also help out in scenarios where the title of the
338                     // RSS feed contains large content
339                     if(description.trim().length() > 0) {
340                         item.setDescription(description);
341                     } else if(title.length() > 0) {
342                     	item.setDescription(title);
343                     } else {
344                     	item.setDescription("");
345                     }
346                     
347                     item.setLink(link);
348                     item.setGuid(guid);
349                     item.setGuidIsPermaLink(guidIsPermaLink);
350                     item.setComments(comments);
351                     item.setCreator(dcCreator);
352 
353                     if (pubDate == null) {
354                         item.setDate(retrievalDate.getTime());
355                         // Add 1 second - to introduce some distinction date-wise
356                         // between items
357                         retrievalDate.add(Calendar.SECOND, 1);
358                     } else {
359                         item.setDate(pubDate);
360                     }
361 
362 					// Invoke ItemProcessor
363 					invokeProcessors(item);
364 
365                     // persist to database...
366                     channelDAO.saveItem(item);
367                     channel.setTotalArticles(channel.getTotalArticles() + 1);
368                 }
369             }
370 
371         }
372 
373         if (!keepHistory) {
374             if (currentSignatures.size() > 0) {
375                 if (channel.getExpiration() == 0) {
376                     channelDAO.deleteItemsNotInSet(channel, currentSignatures);
377 					channel.setTotalArticles(currentSignatures.size());
378                 } else if (
379                     channel.getExpiration() > 0
380                         && (channel.getLastCleaned() == null || channel.getLastCleaned().before(
381                             new Date(
382                                 System.currentTimeMillis()
383                                     - Channel.CLEANING_INTERVAL)))) {
384                     channelDAO.deleteExpiredItems(channel, currentSignatures);
385                     channel.setLastCleaned(new Date());
386                 }
387             }
388         }
389     }

391     @Override
392     public String processContent(Element itemElm) {
393         // Check for xhtml:body
394         String description = null;
395 
396         NodeList bodyList =
397             itemElm.getElementsByTagNameNS(RSSHelper.XMLNS_XHTML, "body");
398         if (bodyList.getLength() > 0) {
399             Node bodyElm = bodyList.item(0);
400             NodeList children = bodyElm.getChildNodes();
401             StringBuffer content = new StringBuffer();
402             for (int childCount = 0;
403                 childCount < children.getLength();
404                 childCount++) {
405                 content.append(children.item(childCount).toString());
406             }
407             description = content.toString();
408         }
409 
410         // Fix for content:encoded section of RSS 1.0/2.0
411         if ((description == null) || (description.length() == 0)) {
412             description =
413                 XMLHelper.getChildElementValue(itemElm, "content:encoded");
414         }
415 
416         if ((description == null) || (description.length() == 0)) {
417             description =
418                 XMLHelper.getChildElementValue(itemElm, "description", "");
419         }
420         return description;
421     }
