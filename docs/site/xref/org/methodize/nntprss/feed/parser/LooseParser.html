50   * 'Loose' Parser - when enabled, will parse those
51   * not well-formed RSS/RDF xml documents on which the 
52   * standard XML parser chokes.
53   * 
54   * This first version is an interesting kludge based 
55   * upon the HTML parsing capabilities available within
56   * Swing.
57   * 
58   * TODO: More tests on internationalization
59   *

61  public class LooseParser {
62  
63      private static DTD dtd = null;
64  
65      static {
66          try {
67              dtd = DTD.getDTD("html32");
68          } catch (IOException ie) {
69          }
70  
71          dtd.getElement("description");
72          javax.swing.text.html.parser.Element element = dtd.getElement("rss");
73          element.getAttribute("version");
74  
75          dtd.getElement("rdf");
76          dtd.getElement("channel");
77          dtd.getElement("category");
78          dtd.getElement("link");
79          dtd.getElement("language");
80          dtd.getElement("title");
81          dtd.getElement("admin");
82          dtd.getElement("item");
83  
84          dtd.defEntity("lt", DTDConstants.GENERAL, '<');
85          dtd.defEntity("gt", DTDConstants.GENERAL, '>');
86          dtd.defEntity("nbsp", DTDConstants.GENERAL, ' ');
87          dtd.defEntity("amp", DTDConstants.GENERAL, '&');
88          dtd.defEntity("quot", DTDConstants.GENERAL, '"');
89          dtd.defEntity("apos", DTDConstants.GENERAL, '\'');
90      }

92      public static Document parse(InputStream is)
93          throws IOException, ParserConfigurationException {
94  
95          DocumentBuilder db = AppConstants.newDocumentBuilder();
96  
97          Document doc = db.newDocument();
98          Element rootElm = doc.createElement("rss");
99          doc.appendChild(rootElm);
100         Element channelElm = doc.createElement("channel");
101         rootElm.appendChild(channelElm);
102 
103         Reader reader = new InputStreamReader(is);
104 
105         CharArrayWriter caw = new CharArrayWriter();
106 
107         // A little hack to use Swing's parser
108         // This informs the parser that the content is within
109         // the body, and therefore all text should be passed
110         // to the callback.
111 
112         // TODO: Extract XML PI 
113         caw.write("<html><body>");
114         char[] buf = new char[1024];
115         int charsRead = reader.read(buf);
116         while (charsRead > -1) {
117             if (charsRead > 0) {
118                 caw.write(buf, 0, charsRead);
119             }
120             charsRead = reader.read(buf);
121         }
122         caw.write("</body></html>");
123         caw.flush();
124         caw.close();
125         reader.close();
126 
127         reader = new CharArrayReader(caw.toCharArray());
128 
129         DocumentParser docParser = new DocumentParser(dtd);
130         docParser.parse(reader, new ParserCallback(doc), false);
131 
132         reader.close();
133 
134         return doc;
135 
136     }
