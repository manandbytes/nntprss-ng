80      /**
81       * @see org.methodize.nntprss.feed.publish.Publisher#publish(Map, String)
82       */
83      public void publish(Map profile, Item content) throws PublisherException {
84  
85          // LJ.XMLRPC.postevent
86          //
87          try {
88              XmlRpcClient xmlrpc =
89                  new XmlRpcClient((String) profile.get(PROP_PUBLISHER_URL));
90              Vector params = new Vector();
91  
92              Map struct = new Hashtable();
93  
94              // username (string): Login for a Blogger user who has permission to post to the blog. 
95              struct.put(STRUCT_USERNAME, profile.get(PROP_USERNAME));
96              // password (string): Password for said username. 
97              struct.put(STRUCT_PASSWORD, profile.get(PROP_PASSWORD));
98              // LiveJournal API version
99              struct.put(STRUCT_VER, "1");
100             // Event - content of the post
101             struct.put(STRUCT_EVENT, content.getDescription());
102             // Line Endings - set to PC (default \r\n)
103             struct.put(STRUCT_LINEENDINGS, "pc");
104             // Subject / Title
105             struct.put(STRUCT_SUBJECT, content.getTitle());
106 
107             Calendar cal = Calendar.getInstance();
108             cal.setTimeInMillis(System.currentTimeMillis());
109 
110             struct.put(STRUCT_YEAR, new Integer(cal.get(Calendar.YEAR)));
111             struct.put(STRUCT_MON, new Integer(cal.get(Calendar.MONTH) + 1));
112             struct.put(STRUCT_DAY, new Integer(cal.get(Calendar.DAY_OF_MONTH)));
113             struct.put(STRUCT_HOUR, new Integer(cal.get(Calendar.HOUR_OF_DAY)));
114             struct.put(STRUCT_MIN, new Integer(cal.get(Calendar.MINUTE)));
115 
116             params.add(struct);
117 
118             /* Map response = (Map) */ xmlrpc.execute(METHOD_POSTEVENT, params);
119 
120             // Discard response
121             // anum - The key number used to calculate the public itemid ID number for URLs.
122             // itemid - The unique number the server assigned to the post.
123 
124         } catch (Exception e) {
125             throw new PublisherException(e);
126         }
127     }

146             // LiveJournal API version

151                 "Java-nntprss/" + AppConstants.VERSION);
152 
153             params.add(struct);
154 
155             /* Map userInfo = (Map) */ xmlrpc.execute(METHOD_LOGIN, params);
