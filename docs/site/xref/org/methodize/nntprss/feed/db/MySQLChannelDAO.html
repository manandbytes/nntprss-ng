209     protected void populateInitialChannels(Document config) {
210         Connection conn = null;
211         PreparedStatement ps = null;
212 
213 		if (!migrateHsql()) {
214 
215             if (log.isInfoEnabled()) {
216                 log.info("Loading channels");
217             }
218 
219             try {
220                 conn =
221                     DriverManager.getConnection(
222                         JdbcChannelDAO.POOL_CONNECT_STRING);
223                 NodeList channelsList =
224                     config.getDocumentElement().getElementsByTagName(
225                         "channels");
226 
227                 if (channelsList.getLength() > 0) {
228                     Element channelsElm = (Element) channelsList.item(0);
229                     NodeList channelList =
230                         channelsElm.getElementsByTagName("channel");
231                     if (channelList.getLength() > 0) {
232 
233                         ps =
234                             conn.prepareStatement(
235                                 "INSERT INTO "
236                                     + TABLE_CHANNELS
237                                     + "(url, name, created, lastPolled, lastArticle, enabled, postingEnabled, parseAtAllCost, pollingInterval, status, expiration) "
238                                     + "values(?, ?, ?, 0, ?, "
239                                     + MYSQL_TRUE
240                                     + ", "
241                                     + MYSQL_FALSE
242                                     + ", "
243                                     + MYSQL_FALSE
244                                     + ", 0, "
245                                     + Channel.STATUS_OK
246                                     + ", "
247                                     + 
248                                     + Channel.EXPIRATION_KEEP
249                                     + ")");
250 
251                         for (int channelCount = 0;
252                             channelCount < channelList.getLength();
253                             channelCount++) {
254                             Element channelElm =
255                                 (Element) channelList.item(channelCount);
256                             String url = channelElm.getAttribute("url");
257                             String name = channelElm.getAttribute("name");
258 
259                             int paramCount = 1;
260                             ps.setString(paramCount++, url);
261                             ps.setString(paramCount++, name);
262                             ps.setTimestamp(
263                                 paramCount++,
264                                 new Timestamp(System.currentTimeMillis()));
265                             // Last Article
266                             ps.setInt(paramCount++, 0);
267                             ps.executeUpdate();
268                         }
269                     }
270                 }
271 
272             } catch (SQLException se) {
273 
274                 if (log.isEnabledFor(Priority.ERROR)) {
275                     log.error("Error loading initial channels", se);
276                 }
277                 throw new RuntimeException(
278                     "Error loading initial channels - " + se.getMessage());
279 
280             } finally {
281                 try {
282                     if (ps != null)
283                         ps.close();
284                 } catch (Exception e) {
285                 }
286                 try {
287                     if (conn != null)
288                         conn.close();
289                 } catch (Exception e) {
290                 }
291             }
292 
293             if (log.isInfoEnabled()) {
294                 log.info("Finished loading initial channels");
295             }
296         }

513     public void addChannel(Channel channel) {
514         Connection conn = null;
515         PreparedStatement ps = null;
516         ResultSet rs = null;
517         Statement s = null;
518 
519         try {
520             conn =
521                 DriverManager.getConnection(JdbcChannelDAO.POOL_CONNECT_STRING);
522             ps =
523                 conn.prepareStatement(
524                     "INSERT INTO "
525                         + TABLE_CHANNELS
526                         + "(url, name, lastArticle, created, enabled, postingEnabled, publishAPI, publishConfig, parseAtAllCost, pollingInterval, status, expiration) "
527                         + "values(?, ?, 0, ?, ?, ?, ?, ?, ?, ?, ?, ?)");
528 
529             int paramCount = 1;
530             ps.setString(paramCount++, channel.getUrl());
531             ps.setString(paramCount++, trim(channel.getName(), FIELD_CHANNEL_NAME_LENGTH));
532             ps.setTimestamp(
533                 paramCount++,
534                 new Timestamp(channel.getCreated().getTime()));
535             ps.setBoolean(paramCount++, channel.isEnabled());
536             ps.setBoolean(paramCount++, channel.isPostingEnabled());
537             ps.setString(paramCount++, channel.getPublishAPI());
538             ps.setString(
539                 paramCount++,
540                 XMLHelper.stringMapToXML(channel.getPublishConfig()));
541             ps.setBoolean(paramCount++, channel.isParseAtAllCost());
542             ps.setLong(paramCount++, channel.getPollingIntervalSeconds());
543             ps.setInt(paramCount++, channel.getStatus());
544             ps.setLong(paramCount++, channel.getExpiration());
545             ps.executeUpdate();
546 
547             ps.close();
548             ps = null;
549 
550             s = conn.createStatement();
551             rs = s.executeQuery("SELECT LAST_INSERT_ID()");
552 
553             if (rs != null) {
554                 if (rs.next()) {
555                     channel.setId(rs.getInt(1));
556                 }
557             }
558         } catch (SQLException se) {
559             throw new RuntimeException(se);
560         } finally {
561             try {
562                 if (ps != null)
563                     ps.close();
564             } catch (SQLException se) {
565             }
566             try {
567                 if (s != null)
568                     s.close();
569             } catch (SQLException se) {
570             }
571             try {
572                 if (conn != null)
573                     conn.close();
574             } catch (SQLException se) {
575             }
576         }
