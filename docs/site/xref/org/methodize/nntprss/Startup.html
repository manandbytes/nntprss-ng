68      public static void main(String[] args) {
69          if (args.length == 0 || !args[0].equals("stop")) {
70              try {
71                  new Startup().run(args);
72              } catch (Exception e) {
73                  e.printStackTrace();
74              }
75          } else {
76              System.exit(0);
77          }
78      }

80      public void run(String[] args) throws Exception {
81          URL resourceURLs[] = new URL[resources.length];
82  
83          for (int resCount = 0; resCount < resources.length; resCount++) {
84              try {
85                  File file = new File(resources[resCount]);
86                  if (file.exists()) {
87                      resourceURLs[resCount] = file.getCanonicalFile().toURL();
88                  } else {
89                      throw new RuntimeException(
90                          "Missing file: " + file.toString());
91                  }
92              } catch (IOException ie) {
93                  ie.printStackTrace();
94              }
95          }
96  
97          ClassLoader existingClassLoader =
98              Thread.currentThread().getContextClassLoader();
99          URLClassLoader urlClassLoader =
100             URLClassLoader.newInstance(resourceURLs, existingClassLoader);
101         Thread.currentThread().setContextClassLoader(urlClassLoader);
102 
103         Class clazz = urlClassLoader.loadClass("org.methodize.nntprss.Main");
104         Method methods[] = clazz.getMethods();
105         Method mainMethod = null;
106         for (int i = 0; i < methods.length; i++) {
107             if (methods[i].getName().equals("main")) {
108                 mainMethod = methods[i];
109                 break;
110             }
111         }
112 
113         mainMethod.invoke(null, new Object[] { args });
114     }
