44  public final class AppConstants {
45  
46      private static final DocumentBuilderFactory docBuilderFactory;
47      private static final String platform;
48      private static final String userAgent;

50      public static final String NNTPRSS_CONFIGURATION_FILE =
51          "nntprss-config.xml";
52  
53      public static final String NNTPRSS_PUBLISH_CONFIGURATION_FILE =
54          "xml/publish-config.xml";

56      public static final String USERS_CONFIG = "users.properties";
57  
58      public static final String VERSION = "0.5-beta-3";
59  
60      public static final int OPEN_ENDED_RANGE = -1;
61  
62      public static final int CONTENT_TYPE_TEXT = 1;
63      public static final int CONTENT_TYPE_HTML = 2;
64      public static final int CONTENT_TYPE_MIXED = 3;
65      
66      static {
67          // Private constructor... initialize platform string
68          StringBuffer pltfmBuf = new StringBuffer();
69          String osName = System.getProperty("os.name");
70          if (osName != null) {
71              pltfmBuf.append(osName);
72              pltfmBuf.append(' ');
73          }
74  
75          String osVersion = System.getProperty("os.version");
76          if (osVersion != null) {
77              pltfmBuf.append(osVersion);
78              pltfmBuf.append(' ');
79          }
80  
81          String osArch = System.getProperty("os.arch");
82          if (osVersion != null) {
83              pltfmBuf.append(osArch);
84          }
85  
86          platform = pltfmBuf.toString();
87          userAgent =
88              "nntprss/"
89                  + VERSION
90                  + " ("
91                  + platform
92                  + "; http://www.methodize.org/nntprss/)";
93  
94          docBuilderFactory = DocumentBuilderFactory.newInstance();
95          docBuilderFactory.setNamespaceAware(true);
96      }
97  
98      private AppConstants() {
99      }

114     public static String getCurrentHostName() {
115         InetAddress localAddr;
116         String hostName = null;
117         try {
118             localAddr = InetAddress.getLocalHost();
119             localAddr =
120                 InetAddress.getByName(
121                     InetAddress.getLocalHost().getHostAddress());
122             hostName = localAddr.getHostName();
123         } catch (UnknownHostException e) {
124         }
125 
126         // Drop down to localhost (127.0.0.1) if we cannot discover host name
127         if (hostName == null) {
128             hostName = "127.0.0.1";
129         }
130 
131         return hostName;
132     }
