54      public static String getChildElementValue(
55          Element parentElm,
56          String elementName) {
57  
58          String elementValue = null;
59          NodeList elemList = parentElm.getChildNodes();
60          Element elm = null;
61          for (int i = 0; i < elemList.getLength(); i++) {
62              if (elemList.item(i).getNodeName().equals(elementName)) {
63                  elm = (Element) elemList.item(i);
64                  elementValue = getElementValue(elm);
65                  break;
66              }
67          }
68  
69          return elementValue;
70      }

72      public static String getElementValue(Element elm) {
73          String elementValue;
74          NodeList childNodes = elm.getChildNodes();
75          StringBuffer value = new StringBuffer();
76          for (int elemCount = 0;
77              elemCount < childNodes.getLength();
78              elemCount++) {
79  
80              if (childNodes.item(elemCount) instanceof org.w3c.dom.Text) {
81                  value.append(childNodes.item(elemCount).getNodeValue());
82              }
83          }
84          elementValue = value.toString();
85          return elementValue;
86      }

88      public static String getChildElementAttributeValue(
89          Element parentElm,
90          String elementName,
91          String attributeName) {
92  
93          String attributeValue = null;
94          NodeList elemList = parentElm.getElementsByTagName(elementName);
95          if (elemList != null && elemList.getLength() > 0) {
96              // Use the first matching child element
97              Element elm = (Element) elemList.item(0);
98              Attr attribute = elm.getAttributeNode(attributeName);
99              if (attribute != null) {
100                 attributeValue = attribute.getValue();
101             }
102         }
103         return attributeValue;
104     }

164     public static String stripTags(String value) {
165         StringTokenizer strTok = new StringTokenizer(value, "<>", true);
166         StringBuffer strippedString = new StringBuffer();
167         boolean inTag = false;
168         while (strTok.hasMoreTokens()) {
169             String token = strTok.nextToken();
170             if (token.equals("<")) {
171                 inTag = true;
172             } else if (token.equals(">")) {
173                 inTag = false;
174             } else if (!inTag) {
175                 strippedString.append(token);
176             }
177         }
178         return strippedString.toString();
179

182     private static String preprocessMarkup(String value) {
183         StringBuffer trimmedString = new StringBuffer();
184         boolean lastCharSpace = false;
185         for (int c = 0; c < value.length(); c++) {
186             char currentChar = value.charAt(c);
187             if (currentChar == '\n') {
188                 trimmedString.append(currentChar);
189             } else if (currentChar < 32) {
190                 continue;
191             } else if (currentChar == ' ') {
192                 if (!lastCharSpace) {
193                     trimmedString.append(currentChar);
194                     lastCharSpace = true;
195                 }
196             } else {
197                 trimmedString.append(currentChar);
198                 lastCharSpace = false;
199             }
200         }
201         return trimmedString.toString();
202     }

204     public static String stripHtmlTags(String value, boolean footnoteLinks) {
205         // Trim white space... Use html markup (p, br) as line breaks
206         value = preprocessMarkup(value);
207 
208         StringTokenizer strTok = new StringTokenizer(value, "<>\n", true);
209         StringBuffer strippedString = new StringBuffer();
210         boolean inTag = false;
211         boolean inPre = false;
212         boolean startOfLine = true;
213         String lastURL = null;
214         List footnotes = null;
215         if (footnoteLinks) {
216             footnotes = new ArrayList();
217         }
218 
219         while (strTok.hasMoreTokens()) {
220             String token = strTok.nextToken();
221             if (token.equals("<")) {
222                 inTag = true;
223 
224                 // Read entire tag... Tag contents might be split over multiple lines
225                 StringBuffer concatToken = new StringBuffer();
226                 while (strTok.hasMoreTokens()) {
227                     token = strTok.nextToken();
228                     if (token.equals(">")) {
229                         inTag = false;
230                         break;
231                     } else {
232                         if (!token.equals("\n")) {
233                             concatToken.append(token);
234                         }
235                     }
236                 }
237 
238                 token = concatToken.toString();
239 
240                 String upperToken = token.toUpperCase();
241                 if (upperToken.startsWith("A ")) {
242                     int hrefPos = upperToken.indexOf("HREF=");
243                     if (hrefPos > -1) {
244                         int quotePos = hrefPos + 5;
245 
246                         while (quotePos < token.length()
247                             && Character.isWhitespace(token.charAt(quotePos))) {
248                             quotePos++;
249                         }
250 
251                         char quote = upperToken.charAt(quotePos);
252 
253                         int endPos;
254                         if (quote == '"' || quote == '\'') {
255                             // URL wrapped in quotes / apostrophes
256                             endPos = token.indexOf(quote, quotePos + 1);
257                         } else {
258                             // URL not enclosed								
259                             endPos = quotePos + 1;
260                             while (endPos < token.length()
261                                 && !Character.isWhitespace(
262                                     token.charAt(endPos))) {
263                                 endPos++;
264                             }
265                         }
266 
267                         if (endPos != -1) {
268                             lastURL = token.substring(quotePos + 1, endPos);
269                             if (upperToken.endsWith("/")) {
270                                 if (!footnoteLinks) {
271                                     // Changed URL wrap characters from parenthesis to lt / gt to avoid
272                                     // issue with certain newsreaders
273                                     strippedString.append(" <URL:");
274                                     strippedString.append(lastURL);
275                                     strippedString.append('>');
276                                 } else {
277                                     footnotes.add(lastURL);
278                                     strippedString.append('[').append(
279                                         footnotes.size()).append(
280                                         ']');
281                                 }
282                                 lastURL = null;
283                                 startOfLine = false;
284                             }
285                         }
286                     }
287                 } else if (upperToken.startsWith("/A")) {
288                     if (lastURL != null) {
289                         if (!footnoteLinks) {
290                             strippedString.append(" <URL:");
291                             strippedString.append(lastURL);
292                             strippedString.append('>');
293                         } else {
294                             footnotes.add(lastURL);
295                             strippedString.append('[').append(
296                                 footnotes.size()).append(
297                                 ']');
298                         }
299                         lastURL = null;
300                         startOfLine = false;
301                     }
302                 } else if (
303                     upperToken.equals("P")
304                         || upperToken.equals("P/")
305                         || upperToken.equals("P /")
306                         || upperToken.equals("UL")
307                         || upperToken.equals("/UL")) {
308                     strippedString.append("\r\n\r\n");
309                     startOfLine = true;
310                 } else if (
311                     upperToken.equals("BR")
312                         || upperToken.equals("BR/")
313                         || upperToken.equals("BR /")
314                         || upperToken.equals("LI")) {
315                     strippedString.append("\r\n");
316                     startOfLine = true;
317                 } else if (upperToken.equals("PRE")) {
318                     inPre = true;
319                 } else if (upperToken.equals("/PRE")) {
320                     inPre = false;
321                 }
322 
323             } else if (token.equals(">")) {
324                 inTag = false;
325             } else if (token.equals("\n")) {
326                 if (!inTag && !startOfLine) {
327                     if (!inPre) {
328                         strippedString.append(' ');
329                     } else {
330                         strippedString.append('\n');
331                     }
332                 }
333             } else if (!inTag) {
334                 strippedString.append(token);
335                 startOfLine = false;
336             }
337         }
338 
339         if (footnoteLinks && footnotes.size() > 0) {
340             strippedString.append("\r\n\r\n");
341             for (int count = 0; count < footnotes.size(); count++) {
342                 strippedString.append(count + 1).append(". ");
343                 strippedString.append(footnotes.get(count));
344                 strippedString.append("\r\n");
345             }
346         }
347         return strippedString.toString();
348 
349     }

351     public static String escapeString(String value) {
352         StringBuffer escapedString = new StringBuffer();
353         for (int charCount = 0; charCount < value.length(); charCount++) {
354             char c = value.charAt(charCount);
355             switch (c) {
356                 case '&' :
357                     escapedString.append("&amp;");
358                     break;
359                 case '<' :
360                     escapedString.append("&lt;");
361                     break;
362                 case '>' :
363                     escapedString.append("&gt;");
364                     break;
365                 case '\"' :
366                     escapedString.append("&quot;");
367                     break;
368                 case '\'' :
369                     escapedString.append("&apos;");
370                     break;
371                 default :
372                     escapedString.append(c);
373             }
374         }
375         return escapedString.toString();
376     }

384     public static String stringMapToXML(Map stringMap) {
385         String mapXMLResult = null;
386         if (stringMap != null && stringMap.size() > 0) {
387             StringBuffer mapXML = new StringBuffer();
388             mapXML.append("<?xml version='1.0' encoding='UTF-8'?>\n<map>\n");
389 
390             Iterator mapIter = stringMap.entrySet().iterator();
391 
392             while (mapIter.hasNext()) {
393                 Map.Entry entry = (Map.Entry) mapIter.next();
394                 mapXML.append("<entry key='");
395                 mapXML.append(escapeString((String) entry.getKey()));
396                 mapXML.append("' value='");
397                 mapXML.append(escapeString((String) entry.getValue()));
398                 mapXML.append("'/>\n");
399             }
400 
401             mapXML.append("</map>");
402 
403             mapXMLResult = mapXML.toString();
404         } else {
405             mapXMLResult = "";
406         }
407         return mapXMLResult;
408     }

410     public static Map xmlToStringHashMap(String xml) {
411         Map map = new HashMap();
412 
413         if (xml != null && xml.length() > 0) {
414             try {
415                 Document doc =
416                     AppConstants.newDocumentBuilder().parse(
417                         new StringInputStream(xml));
418                 Element rootElm = doc.getDocumentElement();
419                 NodeList entryList = rootElm.getElementsByTagName("entry");
420                 for (int elmCount = 0;
421                     elmCount < entryList.getLength();
422                     elmCount++) {
423                     Element entry = (Element) entryList.item(elmCount);
424                     map.put(
425                         entry.getAttribute("key"),
426                         entry.getAttribute("value"));
427                 }
428             } catch (Exception e) {
429                 // XXX do we need to handle this scenario?			
430             }
431         }
432 
433         return map;
