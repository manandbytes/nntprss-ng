341     public static String unescapeString(String value) {
342         StringBuffer unescapedString = new StringBuffer();
343         StringBuffer charBuf = null;
344 
345         for (int pos = 0; pos < value.length(); pos++) {
346             char c = value.charAt(pos);
347 
348             if (c == '&'
349                 && (pos < value.length() - 3)) { // Process reference...
350                 // if less than three characters left, discard
351                 c = value.charAt(++pos);
352 
353                 boolean numeric = false;
354                 if (c == '#') {
355                     numeric = true;
356                     c = value.charAt(++pos);
357                 }
358 
359                 if (charBuf == null) {
360                     charBuf = new StringBuffer(32);
361                 } else {
362                     charBuf.setLength(0);
363                 }
364 
365                 while (c != ';' && pos < value.length() - 1) {
366                     charBuf.append(c);
367                     c = value.charAt(++pos);
368                 }
369 
370                 if (numeric) {
371                     try {
372                         c = (char) Integer.parseInt(charBuf.toString(), 16);
373                     } catch (NumberFormatException nfe) {
374                         // If we can't process it, just write out the text...
375                         unescapedString.append("&#").append(
376                             charBuf.toString()).append(
377                             ';');
378                     }
379                 } else {
380                     Character unescapedVer =
381                         (Character) escapeMap.get(charBuf.toString());
382                     if (unescapedVer != null) {
383                         unescapedString.append(unescapedVer);
384                     } else {
385                         unescapedString.append("&").append(
386                             charBuf.toString()).append(
387                             ';');
388                     }
389                 }
390             } else if (c == '&' && (pos >= value.length() - 3)) {
391                 // If we're at the end of the string, and there's 3 or less characters,
392                 // then assume that this string has been truncated
393                 break;
394             } else {
395                 unescapedString.append(c);
396             }
397 
398         }
399 
400         return unescapedString.toString();
401     }

403     public static String escapeString(String value) {
404         StringBuffer escapedString = new StringBuffer();
405         for (int charCount = 0; charCount < value.length(); charCount++) {
406             char c = value.charAt(charCount);
407             switch (c) {
408                 case '&' :
409                     escapedString.append("&amp;");
410                     break;
411                 case '<' :
412                     escapedString.append("&lt;");
413                     break;
414                 case '>' :
415                     escapedString.append("&gt;");
416                     break;
417                 case '\"' :
418                     escapedString.append("&quot;");
419                     break;
420                 case '\'' :
421                     escapedString.append("&#39;");
422                     break;
423                 default :
424                     escapedString.append(c);
425             }
426         }
427         return escapedString.toString();
428     }

430     public static String stripCRLF(String value) {
431         StringBuffer strippedString = new StringBuffer();
432         for (int i = 0; i < value.length(); i++) {
433             char c = value.charAt(i);
434             if (!(c == '\n' || c == '\r')) {
435                 strippedString.append(c);
436             }
437         }
438         return strippedString.toString();
