211     /**
212      * Encodes up to three bytes of the array <var>source</var>
213      * and writes the resulting four Base64 bytes to <var>destination</var>.
214      * The source and destination arrays can be manipulated
215      * anywhere along their length by specifying 
216      * <var>srcOffset</var> and <var>destOffset</var>.
217      * This method does not check to make sure your arrays
218      * are large enough to accomodate <var>srcOffset</var> + 3 for
219      * the <var>source</var> array or <var>destOffset</var> + 4 for
220      * the <var>destination</var> array.
221      * The actual number of significant bytes in your array is
222      * given by <var>numSigBytes</var>.
223      *
224      * @param source the array to convert
225      * @param srcOffset the index where conversion begins
226      * @param numSigBytes the number of significant bytes in your array
227      * @param destination the array to hold the conversion
228      * @param destOffset the index where output will be put
229      * @return the <var>destination</var> array
230      * @since 1.3
231      */
232     private static byte[] encode3to4(
233         byte[] source,
234         int srcOffset,
235         int numSigBytes,
236         byte[] destination,
237         int destOffset) {
238         //           1         2         3  
239         // 01234567890123456789012345678901 Bit position
240         // --------000000001111111122222222 Array position from threeBytes
241         // --------|    ||    ||    ||    | Six bit groups to index ALPHABET
242         //          >>18  >>12  >> 6  >> 0  Right shift necessary
243         //                0x3f  0x3f  0x3f  Additional AND
244 
245         // Create buffer with zero-padding if there are only one or two
246         // significant bytes passed in the array.
247         // We have to shift left 24 in order to flush out the 1's that appear
248         // when Java treats a value as negative that is cast from a byte to an int.
249         int inBuff =
250             (numSigBytes > 0 ? ((source[srcOffset] << 24) >>> 8) : 0)
251                 | (numSigBytes > 1 ? ((source[srcOffset + 1] << 24) >>> 16) : 0)
252                 | (numSigBytes > 2 ? ((source[srcOffset + 2] << 24) >>> 24) : 0);
253 
254         switch (numSigBytes) {
255             case 3 :
256                 destination[destOffset] = ALPHABET[(inBuff >>> 18)];
257                 destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
258                 destination[destOffset + 2] = ALPHABET[(inBuff >>> 6) & 0x3f];
259                 destination[destOffset + 3] = ALPHABET[(inBuff) & 0x3f];
260                 return destination;
261 
262             case 2 :
263                 destination[destOffset] = ALPHABET[(inBuff >>> 18)];
264                 destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
265                 destination[destOffset + 2] = ALPHABET[(inBuff >>> 6) & 0x3f];
266                 destination[destOffset + 3] = EQUALS_SIGN;
267                 return destination;
268 
269             case 1 :
270                 destination[destOffset] = ALPHABET[(inBuff >>> 18)];
271                 destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
272                 destination[destOffset + 2] = EQUALS_SIGN;
273                 destination[destOffset + 3] = EQUALS_SIGN;
274                 return destination;
275 
276             default :
277                 return destination;
278         } // end switch
279     } // end encode3to4

306     /**
307      * Encodes a byte array into Base64 notation.
308      *
309      * @param source The data to convert
310      * @param off Offset in array where conversion should begin
311      * @param len Length of data to convert
312      * @param breakLines Break lines at 80 characters or less.
313      * @since 1.4
314      */
315     public static String encodeBytes(
316         byte[] source,
317         int off,
318         int len,
319         boolean breakLines) {
320         int len43 = len * 4 / 3;
321             byte[] outBuff = new byte[(len43) // Main 4:3
322         + ((len % 3) > 0 ? 4 : 0) // Account for padding
323     + (breakLines ? (len43 / MAX_LINE_LENGTH) : 0)]; // New lines      
324         int d = 0;
325         int e = 0;
326         int len2 = len - 2;
327         int lineLength = 0;
328         for (; d < len2; d += 3, e += 4) {
329             encode3to4(source, d + off, 3, outBuff, e);
330 
331             lineLength += 4;
332             if (breakLines && lineLength == MAX_LINE_LENGTH) {
333                 outBuff[e + 4] = NEW_LINE;
334                 e++;
335                 lineLength = 0;
336             } // end if: end of line
337         } // en dfor: each piece of array
338 
339         if (d < len) {
340             encode3to4(source, d + off, len - d, outBuff, e);
341             e += 4;
342         } // end if: some padding needed
343 
344         return new String(outBuff, 0, e);
345     } // end encodeBytes
