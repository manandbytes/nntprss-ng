98      public WindowsSysTray() {
99  
100         // TODO icon in the .ico format not visible under Linux
101         Image iconImage = Toolkit.getDefaultToolkit().getImage(ICON_FILE);
102         nntpIcon = new TrayIcon(iconImage, TOOL_TIP, menu);
103         nntpIcon.addActionListener(this);
104         nntpIcon.addMouseListener(this);
105 
106         itemRepoll = new MenuItem(MENU_REPOLL_TEXT);
107         itemRepoll.setActionCommand(MENU_REPOLL_CMD);
108         itemRepoll.setEnabled(false);
109         itemRepoll.addActionListener(this);
110 
111         itemCfg = new MenuItem(MENU_PROPERTIES_TEXT);
112         itemCfg.setActionCommand(MENU_PROPERTIES_CMD);
113         itemCfg.setEnabled(false);
114         itemCfg.addActionListener(this);
115 
116         itemExit = new MenuItem(MENU_EXIT_TEXT);
117         itemExit.setActionCommand(MENU_EXIT_CMD);
118         itemExit.addActionListener(this);
119 
120         itemAbout = new MenuItem(MENU_ABOUT_TEXT);
121         itemAbout.setActionCommand(MENU_ABOUT_CMD);
122         itemAbout.addActionListener(this);
123 
124         menu.add(itemExit);
125         menu.addSeparator();
126         menu.add(itemAbout);
127         menu.addSeparator();
128         menu.add(itemCfg);
129         menu.add(itemRepoll);
130 
131         try {
132             SystemTray tray = SystemTray.getSystemTray();
133             tray.add(nntpIcon);
134             nntpIcon.displayMessage("", "nntp//rss v" + AppConstants.VERSION
135                     + " starting...", MessageType.INFO);
136             UIManager.setLookAndFeel(UIManager.getSystemLookAndFeelClassName());
137         } catch (AWTException e) {
138             System.err.println("TrayIcon could not be added.");
139         } catch (Exception e) {
140             e.printStackTrace();
141         }
142     }

159     private void startBrowser(String url) {
160         try {
161             URI uriToBrowse = new URI(url);
162             if (Desktop.isDesktopSupported()
163                     && Desktop.getDesktop().isSupported(Desktop.Action.BROWSE))
164                 Desktop.getDesktop().browse(uriToBrowse);
165         } catch (IOException e) {
166             // TODO Auto-generated catch block
167             e.printStackTrace();
168         } catch (URISyntaxException e) {
169             // TODO Auto-generated catch block
170             e.printStackTrace();
171         }
172     }

198     @Override
199     public void actionPerformed(ActionEvent e) {
200         if (e == null)
201             return;
202         String actionCommand = e.getActionCommand();
203         if (actionCommand.equals(MENU_ABOUT_CMD)) {
204             final JLabel url =
205                 new JLabel("<html><u><font color=blue>http://www.methodize.org/nntprss</font></u></html>");
206             url.addMouseListener(new MouseListener() {
207                 /**
208                  * @see java.awt.event.MouseListener#mouseClicked(MouseEvent)
209                  */
210                 public void mouseClicked(MouseEvent arg0) {
211                     startBrowser("http://www.methodize.org/nntprss");
212                 }
213         
214                 public void mouseEntered(MouseEvent arg0) {
215                     url.setText(
216                         "<html><u><font color=red>http://www.methodize.org/nntprss</font></u></html>");
217                 }
218         
219                 public void mouseExited(MouseEvent arg0) {
220                     url.setText(
221                         "<html><u><font color=blue>http://www.methodize.org/nntprss</font></u></html>");
222                 }
223         
224                 public void mousePressed(MouseEvent arg0) {
225                 }
226                 public void mouseReleased(MouseEvent arg0) {
227                 }
228         
229             });
230         
231             Object[] message =
232                 new Object[] {
233                     "nntp//rss v" + AppConstants.VERSION,
234                     url,
235                     "\n",
236                     "Copyright (c) 2002-2004 Jason Brome",
237                     "Licensed under the GNU Public License\n" };
238             JOptionPane.showMessageDialog(
239                 null,
240                 message,
241                 MENU_ABOUT_TEXT,
242                 JOptionPane.INFORMATION_MESSAGE);
243         } else if (actionCommand.equals(MENU_EXIT_CMD)) {
244             Object[] options = { "Shutdown", "Cancel" };
245             int option =
246                 JOptionPane.showOptionDialog(
247                     null,
248                     "Are you sure you want to shutdown nntp//rss?",
249                     "nntp//rss - Warning",
250                     JOptionPane.DEFAULT_OPTION,
251                     JOptionPane.WARNING_MESSAGE,
252                     null,
253                     options,
254                     options[1]);
255             if (option == 0) {
256                 System.exit(0);
257             }
258         } else if (actionCommand.equals(MENU_PROPERTIES_CMD)) {
259             startBrowser(adminURL);
260         } else if (actionCommand.equals(MENU_REPOLL_CMD)) {
261             channelManager.repollAllChannels();
262         } else {
263             // Invalid command, should not happen
264             JOptionPane.showMessageDialog(null, actionCommand);
265         }
266     }
